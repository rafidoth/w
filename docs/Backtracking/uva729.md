#include<bits/stdc++.h>

int n,h;

void binfuc(int c, string sel){
    if((int)sel.size() == n){
        int count = 0;
        for(char x : sel) if(x== '1') count++;
        if(count == h) cout << sel <<endl;
        return;
    }
    sel+= "0";
    binfuc(c+1,sel);
    sel.pop_back();
    sel+="1";
    binfuc(c+1,sel);
    sel.pop_back();
}
