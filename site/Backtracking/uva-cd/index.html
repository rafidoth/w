    #include<bits/stdc++.h> 
using namespace std;
typedef long long int lli;
#define MOD 1000000007
#define debug(...) fprintf(stderr, __VA_ARGS__), fflush(stderr)


int N;
int k;
vector<int> arr;
vector<int> sel;
int sum  = 0;
int maxsum = 0;
vector<int> maxv;
void ncr(int curr){
    /*
    for(int x : sel) cout << x << " " ;
    cout<<"sum :"  << sum << endl;
    */
    if(sum == N) {
        maxv = sel;
        maxsum = sum;
        return;
    }
    if((int)sel.size() == k){
        return;
    }
    for(int i =curr; i<k;i++){
        if(sum == N ) return;
        if(sum + arr[i] >  N ) {
            continue;
        }
        sel.push_back(arr[i]);
        sum += arr[i];
        if(sum >maxsum){
            maxsum = sum;
            maxv = sel;
        }
        ncr(i+1);
        if(sum ==N) return;
        sel.pop_back();
        sum -= arr[i];
    }
}
